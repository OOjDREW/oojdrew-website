%Rule


multi_address(?personX) :- address(?personX, ?address1 : String), address(?personX, ?address2 : String),
naf( stringEqualIgnoreCase(?address1 : String, ?address2 : String) ).

%Facts


address(Jack, "552 Windsor" : String).
address(Jack, "615 Graham" : String).

multi_address(?person)
